```python
async def agent_turn(agent: Agent, world: World, step: int) -> AgentTurn:
    # 1. 周囲の発話を取得(3 階の法則で filter)
    visible = filter_visible_messages(agent, world.all_messages, world)
    # 2. 認知限界で切る
    remembered = agent.memory.recent(world.cognitive_limit)
    # 3. プロンプト組み立て
    system = build_system_prompt(agent.persona, agent.current_place, world.economy)
    user = build_user_prompt(visible, remembered)
    # 4. LLM 呼び出し
    resp = await llm.chat([
        {"role": "system", "content": system},
        {"role": "user", "content": user},
    ], seed=agent.seed + step, temperature=0.8, num_predict=256)
    # 5. パース(失敗したら沈黙扱い)
    turn = parse_agent_turn(resp["choices"][0]["message"]["content"])
    if turn is None:
        turn = AgentTurn(utterance="", action="stay", target_id=None)
    return turn
```
