# 05_計測ハーネス(疑似コード)

```python
import structlog
import json
from pathlib import Path

class RunLogger:
    def __init__(self, run_id: str, root: str = "logs/runs"):
        self.dir = Path(root) / run_id
        self.dir.mkdir(parents=True, exist_ok=True)
        self.run_id = run_id
        # ファイルハンドルを開きっぱなし(close は __exit__ で)
        self.llm_io = (self.dir / "llm_io.jsonl").open("a", encoding="utf-8")
        self.perf = (self.dir / "perf.jsonl").open("a", encoding="utf-8")
        self.vram = (self.dir / "vram.jsonl").open("a", encoding="utf-8")
        self.events = (self.dir / "events.jsonl").open("a", encoding="utf-8")
        self.emergence = (self.dir / "emergence.jsonl").open("a", encoding="utf-8")

    def log_llm(self, **record):
        record["run_id"] = self.run_id
        self.llm_io.write(json.dumps(record, ensure_ascii=False) + "\n")
        self.llm_io.flush()

    # perf / vram / events / emergence も同様
```
