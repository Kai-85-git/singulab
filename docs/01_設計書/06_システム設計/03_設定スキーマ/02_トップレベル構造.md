# 02_トップレベル構造

```yaml
meta:
  scenario: "bar_fire"        # シナリオ名(ログ整理用)
  seed: 42                    # 乱数シード(必須)
  description: "参考実装と等価なバー + 火事シナリオ"

simulation:
  duration: 100               # ステップ数
  half_space_size: 25         # フィールド: -25 〜 +25
  dt: 1.0                     # 1 ステップあたりの時間刻み(3 階の物理で使う)

# 1階: 物(場所・障害物)
places:
  - name: "left_bar"
    type: "bar"
    center_x: -15
    center_y: 0
    half_size: 5
    capacity: 12
  - name: "right_bar"
    type: "bar"
    center_x: 15
    center_y: 0
    half_size: 5
    capacity: 10

# 2階: 環境(量)
environment:
  fields: []                  # MVP では空。将来: temperature / pheromone 等
  # 例:
  #   - name: "pheromone_food"
  #     initial: 0.0
  #     unit: "arbitrary"
  #     decay_rate: 0.01

# 3階: 物理法則
physics:
  laws: []                    # MVP では空。将来: diffusion / gravity 等
  # 例:
  #   - type: "diffusion"
  #     target: "pheromone_food"
  #     coefficient: 0.2

# イベント(参考実装の fires を汎化)
events:
  - type: "fire"
    name: "fire_1"
    start_step: 35
    params:
      intensity: 0.8
      radius: 15
      center_x: 15
      center_y: 10
  - type: "fire"
    name: "fire_2"
    start_step: 70
    params:
      intensity: 0.5
      radius: 8
      center_x: -10
      center_y: -10

# エージェント
agents:
  num_agents: 20
  communication_radius: 5
  memory_limit: 20
  memory_size: 5
  message_history_limit: 10
  message_context_size: 3

  # ペルソナ定義(参考実装の gender を包含)
  persona:
    fields:
      - name: "gender"
        values: ["male", "female"]
        assignment: "random"    # random / uniform / list

  # 初期配置ルール
  initial_placement:
    avoid_places: true
    distribution: "uniform"     # uniform / clustered

# LLM
llm:
  provider: "ollama"            # ollama / anthropic / openai
  model: "llama3.1:8b"
  base_url: "http://localhost:11434"
  temperature: 0.7
  max_tokens: 2048
  repeat_penalty: 1.1
  repeat_last_n: 128
  min_p: 0.05

# 可視化
visualization:
  save_frames: true
  frame_interval: 1
  output_dir: "output"
  overlays:
    env_heatmap: null           # 2 階フィールド名を指定するとヒートマップ化
    show_communication_radius: true
    show_event_radius: true

# ログ
logging:
  level: "INFO"
  log_file: "simulation.log"
  outputs:
    messages_jsonl: true
    memory_reasoning_jsonl: true
    events_jsonl: true
    metrics_jsonl: true         # 創発指標(後述)
```
