# 04_擬似コード(Orchestrator)

```python
class Simulation:
    def step_simulation(self):
        self.step += 1

        # (0) イベント活性化
        for ev in self.events:
            if not ev.is_active(self.step) and ev.should_activate(self.step):
                ev.activate(self)
                self._log_event(ev)

        # (1) 物理進行
        for law in self.physics_laws:
            law.step(self.environment, dt=self.dt)

        # (2) Agent 状態更新
        for a in self.agents:
            a.update_state(self.places)

        # Phase 1: Message 決定
        message_decisions = []
        for a in self.agents:
            nearby = a.get_nearby_agents(self.agents)
            ps = self.get_place_status(a.current_place) if a.in_place else None
            env = self.environment.readings_at(a.position)
            evs = [ev.perceived_by(a) for ev in self.events if ev.is_active(self.step)]
            evs = [e for e in evs if e is not None]
            md = a.decide_message(ps, env, nearby, evs, self.step)
            message_decisions.append((a, md, nearby))

        # Phase 2: Message 送信
        for a, md, nearby in message_decisions:
            content = md.get("message", "")
            if content and nearby:
                for other in nearby:
                    other.receive_message(a.id, content, step=self.step)
                    self._log_message(a.id, other.id, content, md.get("reasoning", ""))

        # Phase 3: Action 決定
        action_decisions = []
        memory_records = []
        for a, md, nearby in message_decisions:
            ps = self.get_place_status(a.current_place) if a.in_place else None
            env = self.environment.readings_at(a.position)
            evs = [ev.perceived_by(a) for ev in self.events if ev.is_active(self.step)]
            evs = [e for e in evs if e is not None]
            ad = a.decide_action(ps, env, nearby, evs, md.get("message", ""), self.step)
            action_decisions.append((a, ad))
            memory_records.append({
                "step": self.step, "id": a.id,
                "memory": ad.get("memory", ""),
                "reasoning": ad.get("reasoning", "")
            })
        self._log_memory_reasoning_batch(memory_records)

        # Phase 4: 移動
        for a, ad in action_decisions:
            if ad["action"] == "move" and ad["direction"]:
                a.move(ad["direction"])

        # (5)(6)(7)
        for a in self.agents:
            a.update_state(self.places)
        self._collect_stats()
        self._maybe_visualize()
```
